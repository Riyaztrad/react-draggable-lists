## Install

```bash
npm install --save react-draggable-lists
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'react-draggable-lists'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
